## Queues in Python:

    ## defaultdict:
        -- defaultdict(list) is a special kind of dictionary from Python’s collections module that automatically creates an empty list for a key the first time you access it.
        eg: 
            1. with normal dict or map: 
                d = {}
                d['a'].append(1)   # KeyError ❌

                if 'a' not in d:
                    d['a'] = []
                d['a'].append(1)   # This Works ✅

            2. With defaultdict: 
                from collections import defaultdict

                d = defaultdict(list)
                d['a'].append(1)   # works fine ✅
                d['a'].append(2)

                print(d)           # print: {"a": [1, 2]}

## Heaps -> Min and Max heaps
    -- Bydefault Heaps in Python are minHeap.
    -- They are basically arrays under the hood. 

    import heapq

    # under the hood are arrays
    minHeap = []
    heapq.heappush(minHeap, 3)     # TC: O(log n)
    heapq.heappush(minHeap, 5)
    heapq.heappush(minHeap, 1)

    # Min is always at index 0:
    print(minHeap[0]);              #print: 1   TC: O(1)

    # Looping in a heap: 
    while(len(minHeap)):
        print(heapq.heappop(minHeap))    # pop elements from the heap. --> This pops in sorted ordered i.e 1  3  5   TC: O(log n)

    ## Max Meap Doesn't exist in Python
    -- WorkAround: use minHeap and multiple by -1 when we do push & pop
    maxHeap = []
    heapq.heappush(maxHeap, -3)
    heapq.heappush(maxHeap, -2)
    heapq.heappush(maxHeap, -4)

    # the max element is at index 0
    print(-1 * maxHeap[0])             # Print: 4

    # Looping: 
    while len(maxHeap):
        print(-1 * heapq.heappop(maxHeap))  # pop elements from the heap in descending order i.e 4 3 2


    ## Build the heap from some intial values
    arr = [2, 1, 8, 4, 5]
    heapq.heapify(arr)          # TC: O(N)

    while arr:
        print(heapq.heappop(arr))           # this print: 1 2 4 5 8

## Nested functions -> Really helpful in Recursive function.
    -- Nested Function have access to the outer variables.

    def outer(a, b):
        c = "c"

        # defining a inner function -> can access current scope variables.
        def inner():
            return a + b + c
        
        return inner()     # calling the inner function

    print(outer("a", "b"))

NOTE: 
    # Under Nested Functions we can modify objects like arrays but not reassign unless using "nonlocal" keyword.

    def double(arr, val):
        def helper():
            # Modifying the array works
            for i, n in enumerate(arr):
                arr[i] *= 2

            # The below line -> it will only modify val in the helper scope.
            # val *= 2

            # this will modify val outside the helper scope
            nonlocal val
            val *= 2
        
        helper()
        print(arr, val)

    nums = [1, 2]
    val = 3
    print(double(nums, val))   # print: [2, 4] 6   i.e the array will modify but for the variable's modification to refelect we declare it "nonlocal"
